```

var b64=String(src).match(/ttf;base64,([^\)]+)/);
if (b64) {
    var f1 = java.queryBase64TTF(b64[1])
    var f2 = java.queryTTF("/storage/emulated/0/Fonts/Source Han Sans CN Regular.ttf")
    java.replaceFont(result, f1, f2)
}else{
    result
}
```
